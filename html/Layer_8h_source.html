162 inline CBoundary* CLayer::GetBoundary(CPoint *pPoint1, CPoint *pPoint2)
163 {
164  if(pPoint1->Z()!=pPoint2->Z()) return NULL;
165 
166  return GetBoundary(pPoint1->X(),pPoint1->Y(),pPoint2->X(),pPoint2->Y());

169 inline CBoundary* CLayer::GetBoundary(int iX1, int iY1, int iX2, int iY2)
170 {
171  if(iX1==iX2&&iY1==iY2) return NULL;
172 
173  if(IsHorizontal()&&iY1==iY2)
174  {
175  assert(abs(iX1-iX2)==1);
176  return GetBoundary(MIN(iX1,iX2),iY1);
177  }
178  else if(IsVertical()&&iX1==iX2)
179  {
180  assert(abs(iY1-iY2)==1);
181  return GetBoundary(iX1,MIN(iY1,iY2));
182  }
183 
184  return NULL;
185 }
