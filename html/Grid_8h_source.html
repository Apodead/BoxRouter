138 inline void CGrid::SetMaze(int iMaze)
139 {
140  switch(iMaze){
141  case MAZE_GRID_FLOODED:
142  *m_pMaze |= MAZE_GRID_FLOODED;
143  m_iHeapPos = -1;
144  break;
145  case MAZE_GRID_NULL:
146  *m_pMaze = MAZE_GRID_NULL;
147  m_iState = MAX_NUMBER;
148  m_pMazeGrid = NULL;
149  m_iHeapPos = -1;
150  break;
151  case MAZE_GRID_SOURCE:
152  *m_pMaze |= MAZE_GRID_SOURCE;
153  *m_pMaze &= ~(MAZE_GRID_TARGET|MAZE_GRID_BRIDGE);
154 // *m_pMaze &= ~MAZE_GRID_BRIDGE;
155  m_pMazeGrid = NULL;
156  //m_iState = 0;
157  //m_iMazePCost= 0;
158  m_iHeapPos = -1;
159  break;
160  case MAZE_GRID_TARGET:
161  *m_pMaze |= MAZE_GRID_TARGET;
162  *m_pMaze &= ~(MAZE_GRID_SOURCE|MAZE_GRID_BRIDGE);
163  //*m_pMaze &= ~MAZE_GRID_SOURCE;
164  //*m_pMaze &= ~MAZE_GRID_BRIDGE;
165 // m_iState = MAX_NUMBER;
166  m_pMazeGrid = NULL;
167  m_iHeapPos = -1;
168  break;
169  case MAZE_GRID_BRIDGE:
170  *m_pMaze |= MAZE_GRID_BRIDGE;
171  *m_pMaze &= ~(MAZE_GRID_TARGET|MAZE_GRID_SOURCE);
172  m_pMazeGrid = NULL;
173  //*m_pMaze &= ~MAZE_GRID_SOURCE;
174  //*m_pMaze &= ~MAZE_GRID_TARGET;
175  m_iState = MAX_NUMBER;
176  m_iHeapPos = -1;
177  break;
178  case MAZE_GRID_BOUND:
179  if(*m_pMaze==MAZE_GRID_NULL) *m_pMaze |= MAZE_GRID_BOUND;
180  break;
181  default:
182  assert(FALSE);
183  break;
184  }
185 }
